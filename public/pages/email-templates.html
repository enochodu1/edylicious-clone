Email Integration Options

Option 1: Formspree (Easiest)

// Sign up at formspree.io and get your form ID
const EMAIL_ENDPOINT = 'https://formspree.io/f/YOUR_FORM_ID';

// Send email
fetch(EMAIL_ENDPOINT, {
    method: 'POST',
    headers: {
        'Content-Type': 'application/json',
    },
    body: JSON.stringify({
        email: formData.email,
        _replyto: formData.email,
        _subject: `Booking ${bookingId}`,
        message: emailContent
    })
});

Option 2: EmailJS

// Sign up at emailjs.com
emailjs.init("YOUR_PUBLIC_KEY");

emailjs.send("YOUR_SERVICE_ID", "YOUR_TEMPLATE_ID", {
    to_email: "info@edylicious.com",
    from_email: formData.email,
    booking_details: formData
});

Option 3: Backend API

// Create a simple Node.js/Express endpoint
app.post('/api/booking', async (req, res) => {
    const booking = req.body;
    
    // Save to database
    await saveBooking(booking);
    
    // Send emails
    await sendEmail({
        to: booking.email,
        template: 'booking-confirmation',
        data: booking
    });
    
    await sendEmail({
        to: 'info@edylicious.com',
        template: 'booking-notification',
        data: booking
    });
    
    res.json({ success: true, bookingId: booking.bookingId });
});
